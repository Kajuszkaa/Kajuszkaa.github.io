    <script>
        // Twoje hasła (9 sztuk) + moje szkolne zapychacze, żeby dobić do 24
        const phrases = [
            "Energetyk w szkole", "Marta otwiera morde", "ide z ostatniej", "Spóźnie się+",
            "Jebana Sekretarka", "rzep", "Drama", "Grajek", 
            "Sprawdzian odwołany", "Głupie pytanie z sali", "Ktoś gra na telefonie", "Nauczyciel opowiada o życiu",
            "Awaria projektora", "Brak zadania domowego", "Dzwonek na przerwę za wcześnie", "Kawa z automatu",
            "Panika przed testem", "Ucieczka z wuefu", "Plotki w toalecie", "Problem z Librus",
            "Zastępstwo", "Głód na 3 lekcji", "Ktoś śpi", "Zgubiony długopis"
        ];

        // Zapisywanie planszy i stempli w localStorage
        let savedBoard = JSON.parse(localStorage.getItem('graTerenowaBoard'));
        let savedStamps = JSON.parse(localStorage.getItem('graTerenowaStamps')) || {};

        if (!savedBoard) {
            phrases.sort(() => Math.random() - 0.5);
            savedBoard = phrases;
            localStorage.setItem('graTerenowaBoard', JSON.stringify(savedBoard));
        }

        const board = document.getElementById('board');
        let phraseIndex = 0;

        // Tworzymy 25 kratek
        for (let i = 0; i < 25; i++) {
            const square = document.createElement('div');
            square.className = 'square';

            // Środkowe pole to zawsze JACKPOT
            if (i === 12) {
                square.classList.add('center');
                square.innerHTML = `
                    <div class="center-star">⭐</div>
                    <span class="square-text">JACKPOT</span>
                `;
            } else {
                square.innerHTML = `<span class="square-text">${savedBoard[phraseIndex]}</span>`;
                phraseIndex++;
            }

            // Odtwarzanie zapisanych stempli po odświeżeniu
            if (savedStamps[i]) {
                const stamp = document.createElement('div');
                stamp.className = 'stamp';
                stamp.innerText = savedStamps[i].emoji;
                stamp.style.transform = `rotate(${savedStamps[i].rotation}deg)`;
                square.appendChild(stamp);
            }

            // Funkcja dodająca "stempel" (emotkę) gracza po kliknięciu
            square.addEventListener('click', function() {
                // Sprawdzamy, czy na tym polu już jest stempel
                const existingStamp = this.querySelector('.stamp');
                
                if (existingStamp) {
                    // Jeśli jest, to kliknięcie go usuwa (cofnięcie zaznaczenia)
                    existingStamp.remove();
                    
                    // Usuwamy też z pamięci
                    delete savedStamps[i];
                    localStorage.setItem('graTerenowaStamps', JSON.stringify(savedStamps));
                } else {
                    // Jeśli nie ma, pobieramy wybraną emotkę z listy...
                    const selectedEmoji = document.getElementById('playerSelect').value;
                    // ...i dodajemy ją na środek pola
                    const stamp = document.createElement('div');
                    stamp.className = 'stamp';
                    stamp.innerText = selectedEmoji;
                    
                    // Lekko losowy obrót stempla dla naturalnego efektu naklejki
                    const randomRotation = Math.floor(Math.random() * 30) - 15;
                    stamp.style.transform = `rotate(${randomRotation}deg)`;
                    
                    this.appendChild(stamp);

                    // Zapisujemy w pamięci
                    savedStamps[i] = { emoji: selectedEmoji, rotation: randomRotation };
                    localStorage.setItem('graTerenowaStamps', JSON.stringify(savedStamps));
                }
            });

            board.appendChild(square);
        }

        // Obsługa przycisku resetowania gry
        document.getElementById('resetBtn').addEventListener('click', () => {
            if (confirm('Czy na pewno chcesz wyczyścić planszę i wylosować nowe hasła?')) {
                localStorage.removeItem('graTerenowaBoard');
                localStorage.removeItem('graTerenowaStamps');
                location.reload();
            }
        });
    </script>
